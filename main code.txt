<script>
function calculateResult() {
    let scores = {
        visual: 0,
        auditory: 0,
        interpersonal: 0,
        independent: 0
    };

    const answers = document.querySelectorAll('input[type="radio"]:checked');

    if (answers.length < 10) {
        alert("Please answer all questions.");
        return;
    }

    answers.forEach(answer => {
        scores[answer.value]++;
    });

    const maxScore = Math.max(
        scores.visual,
        scores.auditory,
        scores.interpersonal,
        scores.independent
    );

    let dominantStyles = [];

    if (scores.visual === maxScore) dominantStyles.push("Visual Learner 📊");
    if (scores.auditory === maxScore) dominantStyles.push("Auditory Learner 🎧");
    if (scores.interpersonal === maxScore) dominantStyles.push("Interpersonal Learner 🤝");
    if (scores.independent === maxScore) dominantStyles.push("Independent Learner 📚");

    let resultText = "";

    if (dominantStyles.length === 1) {
        resultText = "Your dominant cognitive learning style is:<br><br>" +
                     dominantStyles[0];
    } else {
        resultText = "You have multiple dominant cognitive learning styles:<br><br>" +
                     dominantStyles.join("<br>");
    }

    document.getElementById("result").innerHTML = resultText;
}
</script>
